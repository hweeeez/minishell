typedef enum e_node_type
{
    NODE_PIPE,
    NODE_COMMAND,
    NODE_REDIR,
    NODE_ARGS,
} t_node_type;

typedef struct s_tree
{
	t_node*		root;
	t_pipe*		last_pipe;
}				t_tree;

typedef struct s_node
{
	int			type;
	t_pipe*		pipe;
	t_command*	commands;
	t_redir*	redir;
}				t_node;

typedef struct s_pipe
{
	s_node*		left;
	s_node*		right;
}				t_pipe;

typedef struct s_command
{
	char**		args;
	t_redir*	redirs;
}				t_pipe;

typedef struct s_redir
{
	char*		file;
	int			type;
}				t_redir;
